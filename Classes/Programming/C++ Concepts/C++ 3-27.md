# Vectors
- can store multiple items like arrays
- all the same data type
- unlike arrays, you CAN change the size of a vector

```C++

int main() 
{
	int numChildren = 5;
	// vectors 
	vector<int> numPiecesCandy(numChildren);
	vector<string> childNames(numChildren);
	int total = 0;
	double average;
	cout << "\nHow many children ate halloween candy? ";
	cin >> numChildren;

	// vector resizing 
	childNames.resize(numChildren);
	numPiecesCandy.resize(numChildren);

	cout << "\nHow many pieces of candy did the children eat on Halloween?"
	for(int i=0; i<numChildren; i++)
	{
		cout << "CHILD " << i+1 << ":\n";
		cout << "\tName - ";
		getline(cin, childNames.at(i));
		cout << "\tPieces of Candy - ";
		cin >> numPiecesCandy.at(i);
	}

	for(int i=0; i<numChildren; i++)
	{
		total += numPiecesCandy.at(i)
	}
	average = static_cast<double>(total) / numChildren;

	cout << "\nThe average amount of candy eaten by the children is << average << endl;"
	
	return 0;
}

```
